הגדרת סוג השלוחה

type=record_system_messages
קישורים לתוספים שניתנים להגדרה בכל מודול

הגדרות זיהוי בכניסה לשלוחה

הרשאות כניסה לשלוחה
ברירת מחדל מבקש לבחור באיזה תיקיה להקליט
ואיזה שם קובץ של הודעת מערכת

באם ברצונכם להגדיר מראש את השלוחה ליצירת הודעת מערכת = ניתן להגדיר את מספר השלוחה

record_system_folder=/8/8
באם ברצונכם להגדיר יצירת הודעת מערכת, בשלוחה הראשית ניתן להגדיר

record_system_folder=main
באם ברצונכם להגדיר יצירת הודעת מערכת, בשלוחה של כל הודעות ברירת מחדל ברמת מערכת, ניתן להגדיר

record_system_folder=/Messages

record_system_messages=M1000
