ברירת מחדל במערכת ניתן להעתיק תכנים למערכות אחרות. העתקה מתאפשרת רק למערכות שבאותו שרת | כיום יש ארבעה שרתים 1. ymtA .4| ymtB .3| yemot .2 | ymt

לשינוי ההגדרה במערכת חברתית מסוג ווצאפפון או בלוג פון וכדומה יש להיכנס לשלוחת הניהול (שלוחה 7) במקש 6.

במערכות ivr2 יש להגדיר

copy_link=no
ניתן להגדיר את זה בשלוחה מסוימת למניעת העתקה מהשלוחה, או בקובץ ivr.ini בשלוחה הראשית בשביל למנוע העתקה מכלל המערכת.

בהעתקת הקובץ למערכות ivr2 ברירת מחדל יש צורך להקיש סיסמת ניהול בשביל להעתיק לשם את הקובץ
ניתן לבטל את הסיסמה או מכולם או מהאנשים שנמצאים בטמפלט ברירת המחדל שלכם
את ההגדרה יש לשים אך ורק בקובץ ivr.ini במערכת היעד של השיתוף

copy_link_no_password=all
copy_link_no_password=template_default
בדוגמה הראשונה כל אחד יוכל להעתיק ובדוגמה השניה רק מי שנמצא בטמפלט ברירת המחדל יוכל להעתיק.

בשביל לשתף קובץ עם מערכת ווצאפון אחרת הנכם צריכים להיות חברים בקבוצה של מערכת היעד

ברירת מחדל ניתן לשתף את הקובץ עם מערכות ivr2 בתנאי שתייקית היעד הינה מסוג השמעת קבצים - type=playfile
אם ברצונכם שניתן יהיה לשתף עם כל שלוחה קיימת במערכת ללא קשר לסוג השלוחה יש להגדיר

copy_link_all_folders=yes
ברירת מחדל ניתן להעתיק לכלל השלוחות במערכת (לפי סוג של השמעת קבצים או לכל השלוחות כמופיע לעיל)
אם רוצים שיהיה ניתן להעתיק רק לשלוחות מסוימות במערכת ניתן להגדיר בקובץ ivr.ini במערכת היעד של השיתוף

copy_link_folders_allowed=1,2,5/5,7/8
בדוגמא הנ"ל יוכל להעתיק רק לתיקיות 1,2,5/5,7/8.

אם ברצונכם שבשיתוף למספר של המערכת הקובץ יועבר ישירות לשלוחה מסויימת ולא יהיה תפריט בחירה לאיזה שלוחה לשתף יש להגדיר בקובץ CopyLinkShareToFolder.ini שבשלוחה הראשית את ההגדרה הבאה

0773137770=/1
0773137777=/2
בדוגמה - אם השיתוף בוצע למספר 0773137770 הקובץ יועבר ישירות לשלוחה 1 ואם השיתוך בוצע למספר 0773137777 הקובץ יועבר ישירות לשלוחה 2.

ניתן גם להגדיר שהמערכת תבצע בדיקה האם המספר מפנה לשלוחה פנימית במערכת וא"כ הקובץ יועבר לשם (ניתן להגדיר בקובץ Did_Go_To.ini)

copy_link_to_folder_in_did_go_to=yes
את ההגדרה יש לשים בקובץ ivr.ini במערכת היעד
