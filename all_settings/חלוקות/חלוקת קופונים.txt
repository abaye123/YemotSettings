type=coupon
קישורים לתוספים שניתנים להגדרה בכל מודול
רשימת כל ההגדרות שניתנות להטמעה בכל מודול
הגדרות זיהוי בכניסה לשלוחה
הסבר על הודעת ברוכים הבאים (הודעה ראשונה בשלוחה)
הרשאות כניסה לשלוחה
הכניסה לתיקיה רק מטלפון מזוהה, אחרת יודיע
M1015 מספרך אינו מזוהה במערכת.

את הקופונים עצמם יש להכניס לקובץ בתוך התיקיה בשם
coupon.ini
בתוך הקובץ עצמו יש לרשום

;BH
1001=erty
1002=vgii
1003=wqer
כלומר קופון 1001 הקוד שלו הוא erty וכו'
המערכת יכול להשמיע קופון המכיל גם ספרות וגם אותיות

במידה והקובץ coupon.ini לא קיים תושמע הודעה
M1414 אין כרגע קופונים לחלוקה

ניתן לשים הודעה ראשונית בשם M0000.wav

המערכת תשמיע את הקופון ותודיע
M1415 הקופון שלך הוא

השמעת הקופון:
בחירת מחדל משמיע את הקופון באותיות אנגלית או ספרות בלבד.
ניתן להגדיר:

coupon_sey_tts=yes
ואז ישמיע את הקופון כפי שהוא בדיוק בTTS, למשל, אם כתוב:

1001=שמואל
יקריא "שמואל" בTTS.
כל ערכי הTTS תקפים, קול, מהירות, וכד'

ברירת מחדל הקופון הראשון לחלוקה הוא מתחיל ממספר 1000+
אא"כ מוגדר אחרת.

coupon_start=0
יש לציין מספר אחד יותר נמוך ממה שיתן ראשון, לדוגמא באם יוגדר 0 המספר הראשון שינתן יהיה 1

בכניסה לתיקיה בודק האם יש הגבלה בערך "קופון מקסימום"

coupon_max=1040
ואז באם הקופון הקיים כרגע גדול מהערך של מקסימום קופונים יודיע הודעה
M1413 הקופונים נגמרו.
וחזור שלב אחורה

על מנת שהאישורים עם קובץ ובו כל האישורים עד היום יעברו בכל פעם לאיימיל יש להגדיר

emeil_send=yes
האיימיל שאליו נשלח - והשם שיופיע באיימיל זה כמוגדר בערכים CustomerEmail CustomerName כלומר מה שכתוב בחייגן
אם רוצים לשנות את האיימיל ניתן להכניס בערכים הבאים
email_address=AAA@gmail.com
email_name=avraem

לאחר השמעת האישור משמיע תפריט
M1416 "לסיום הקישו 1.
לשמיעת הקופון פעם נוספת הקישו 2."

ניתן להגדיר שעם סיום המערכת תעביר את המאזין לשלוחה אחרת או ניתוק | ראה כאן את האפשרויות להגדרה.
