זהירות בשימוש! הגדרה של השלוחה כאשר למאזינים יש גישה אליה, תגרום לכך שכל אחד מהמאזינים יוכל להפעיל את הקמפיין הטלפוני!!
type=yemot_dialer_campaign_start
קישורים לתוספים שניתנים להגדרה בכל מודול
רשימת כל ההגדרות שניתנות להטמעה בכל מודול
הגדרות זיהוי בכניסה לשלוחה
הסבר על הודעת ברוכים הבאים (הודעה ראשונה בשלוחה)
הרשאות כניסה לשלוחה
ברירת מחדל מפעיל את תבנית ברירת המחדל
ניתן להגידר איזה תבנית רוצים להפעיל בקמפיין זה

template_to_start=2
בחירת מחדל מפעיל קמפיין רגיל, אפשר להגדיר הפעלת קמפיין צינתוק:

campaign_run_tzintuk=yes
חיוב היחידות בפועל בקמפיין צינתוק יהיה 0.1 יחידה על כל מספר.
בקמפיין צינתוק ניתן להגדיר כמה זמן יהיה הצינתוק, הזמן המרבי הוא 10 שניות, בחירת מחדל היא 10 שניות:

campaign_run_tzintuk_timeout=5
ברירת מחדל משמיע
ברשותך..ברשימת התפוצה.. עבור קמפינין זה תחוייב..
אם רוצים שהמערכת לא תשמיע אלא תדלג ישר ל"אישור הפעלת הקמפיין הקישו אחת" יש להגדיר

campaign_say_details=no
ברירת מחדל משמיע "אישור הפעלת הקמפיין הקישו אחת"
באם רוצים שיפעיל מיידית את הקמפיין גם ללא בקשת אישור יש להגדיר

campaign_ask_before_start=no
ברירת מחדל שכאשר הקמפיין הראשון עדיין לא הסתיים ומפעילים פעם שניה אז המערכת שואלת

"לקוח יקר הקמפיין שביקשת להפעיל כבר פועל במערכת ועדיין לא הסתיים, האם אתה בטוח שברצונך להפעיל קמפיין נוסף? להפעלת קמפין נוסף הקישו 1 אחרת הקישו 2"

באם רוצים שגם על הפעלת קמפיין נוסף לא ישאל כלום אלא מייד יפעיל אותו יש להגדיר

campaign_ask_before_double_start=no
ברירת מחדל לאחמ"כ חוזר שלב אחד אחורה ניתן להגדיר מעבר לשלוחה אחרת או ניתוק | ראה כאן את האפשרויות להגדרה.

end_goto=/1
