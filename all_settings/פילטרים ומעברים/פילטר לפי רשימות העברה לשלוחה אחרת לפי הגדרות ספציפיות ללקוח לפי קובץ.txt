ההעברה לשלוחה אחרת לפי הגדרה באחת מרשומות הקובץ "ListAllInformation.ini"

סוג השלוחה

type=go_to_folder_from_list_all_information
קישורים לתוספים שניתנים להגדרה בכל מודול
רשימת כל ההגדרות שניתנות להטמעה בכל מודול
הגדרות זיהוי בכניסה לשלוחה
במודול זה הכניסה לפי ID היא חובה | ראה פירוט להלן
הסבר על הודעת ברוכים הבאים (הודעה ראשונה בשלוחה)
הרשאות כניסה לשלוחה
הקובץ שממנו בודקים את הנתונים ListAllInformation.ini שנמצא בתיקיית EnterID

(הסבר על הקובץ כאן)

בכניסה לשלוחה המערכת תבצע כניסה לפי מספר אישי. ברירת מחדל לפי סוג

enter_id_type=list_all_information
ניתן לשנות את סוג הכניסה לפי כל אחד מהסוגים הקיימים והמערכת תשלוף את הנתונים מהקובץ.

הערך הראשון שחובה להגדיר - מאיזה עמודה בטבלה לבדוק את הנתונים. ההגדרה היא:

value_number=8
בדוגמה, המערכת תחפש את מה שכתוב בעמודה 8 באותה שורה של המספר האישי.

בנוסף, יש צורך להגדיר כל ערך שמופיע בעמודה 8 לאיזה תיקיה להפנות.

ניתן להגדיר שילך לתיקייה כשם הערך שהתקבל - כלומר אם מה שמופיע בטבלה הוא aaa המערכת תחפש את שלוחת aaa שחתת התיקייה הנוכחית ותעביר את המתקשר לשם.

ההגדרה היא:

go_to_folder_val=yes
באם יש הגדרה שונה מאשר הערך שהתקבל ניתן להגדיר לאן להפנות כל אחד מהתוצאות

ברירת מחדל - המערכת לוקחת את הערכים לתוצאות מתוך הקובץ ext.ini

ניתן להגדיר

go_to_folder_from_file=yes
והמערכת תיקח את הערכים לתוצאות מקובץ GoToFolderFromListAllInformation.ini שנמצא בשלוחה.

(מומלץ להגדיר את זה מכיוון שכמות הערכים שניתן להגדיר בקובץ ext.ini מוגבלת ל10)

לדוגמה, במידה ומה שמופיע בעמודה 8 זה המילה ALL ואני רוצה שאם כתוב ALL ילך לתיקיה 9 שתחת לתיקיה הנוכחית, יש לכתוב

ALL=9
ואם רוצים שילך לתיקיה 9 שמתחת לתיקיה הראשית יש לכתוב

ALL=/9
כמובן שניתן לשייך כמה תוצאות לשלוחה אחת

לדוגמה

ALL=9
Chabad=9
big=7
min=5
בנוסף, ניתן להגדיר ערך מיוחד לאיזה שלוחה לעבור במידה ואין התייחסות בעמודה 8 בטבלה,
או שמה שמופיע בעמודה 8 אינו תואם שום דבר בהגדרות לאיזה שלוחה לעבור
אז ניתן להגדיר ערך ברירת מחדל לאיפה לעבור בכזה מצב

go_to_folder_default=4
כאן רואים שבמידה ואין הגדרה יעבור לשלוחה 4 שמתחת לשלוחה נוכחית.

בנוסף ניתן להגדיר ערכים מה לעשות אם הערך גדול ממספר או קטן ממספר

יש להכניס את הערכים הבאים (לדוגמה)

type=go_to_folder_from_list_all_information
value_number=11
go_to_folder_if_big=10:big:small
go_to_folder_default=all
הסבר:
value_number=11 - תחפש בעמודה 11 שבטבלה

go_to_folder_if_big=10:big:small - אם מה שכתוב בעמודה 11 המספר הוא גדול מ10 אז תפעיל את תיקיית
big ואם המספר הוא שווה ל10 או קטן מ10 אז תפעיל את תיקיית small שמתחת לתיקיה הנוכחית.

go_to_folder_default=all - אם אין הגדרה או שהפעלה של התיקייה המבוקשת לא קיימת או לא ניתנת להפעלה - אז תפעיל את תיקיה all שמתחת לתיקיה הנוכחית
