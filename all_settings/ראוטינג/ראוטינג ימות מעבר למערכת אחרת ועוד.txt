routing - ניתוב שלוחה (למערכת אחרת, למספר טלפון, לכמה מספרים ביחד, ל1800 לIP)

ישנם כמה סוגים של ניתוב בשלוחות:

1 - ניתוב למערכת אחרת של ימות המשיח (אין חיוב ביחידות)/ ניתוב למערכות אחרות של ימות המשיח לפי הגדרות זמנים (אין חיוב ביחידות)

2 - ניתוב לטלפון אישי לפי תפריט לאיזה שלוחה לצלצל/ לפי הגדרה ישירה לאיזה שלוחה לצלצל / לפי הגדרה של לצלצל לשלוחות לפי סדר עם הגבלת כמות / לצלצל לכמה מספרים ביחד
(חיוב ביחידה לדקה, מדידה לפי שניות)

3 - ניתוב לטלפון אישי לפי שעות / ימים (חיוב ביחידה לדקה, מדידה לפי שניות)

4 - ניתוב למספרי 1800/1801 (אין חיוב ביחידות)

ניתוב למערכת אחרת של ימות המשיח:
type=routing_yemot
routing_yemot_number=0773137770
;באם רוצים לנעול ולשים בקשת סיסמה בכניסה לתיקיה ניתן להכניס ערך זה.

password=1234
; להשמיע את קובץ M0000 רק פעם אחת בכל תיקיה בכל פעם שמתקשר למערכת

say_first_one_time=yes
הודעות
ו - הקובץ המושמע ראשון,לא חובה באם אין - מדלג

חסימת אפשרות ההעברה למערכת שלך:

באם ברצונך שמודול routing_yemot ניתוב פנימי בין המערכות - לא יוכל לעבוד במערכות אחרות שיוכלו להגיע למערכת שלך כניתוב פנימי
יש צורך להגדיר במערכת בתיקיה ראשית בקובץ ivr.ini את ההגדרה (שימו לב, לא בקובץ ext.ini הנמצא בתוך השלוחה).

routing_from_yemot=no
זה גורם שלא יוכלו להפנות שיחות למערכת שלך ממערכות אחרות באמצעות ניתוב פנימי.

הפניה לתוך שלוחה מסוימת במערכת אחרת

(שימו לב! המערכת שאליה מפנים צריכה להיות בניהולכם!)

במערכת שממנה אתם מפנים תגדירו (בשלוחה שתבחרו להפנות ממנו)
type=routing_yemot
routing_yemot_number=079512345
לא לשכוח להחליף את המספר 079512345 למספר המערכת שאליה את מפנים

ובמערכת שאליה אתם מפנים - תגדירו בשלוחה הראשית - בקובץ ההגדרות (ext.ini)

check_did_and_go_to_folder=yes
(שזה "זיהוי לפי מספר מחויג")

ובנוסף להכניס (במערכת שאליה אתם מפנים) - קובץ בשם Did_Go_To.ini
ובו לכתוב

079512345=/4
(לא לשכוח להחליף את המספר 079512345 למספר המערכת שאליה את מפנים)
זה יפנה לשלוחה 4

שימו לב! אם המערכת משמשת אתכם לעוד דברים
אז אתם תצטרכו שתי מספרים למערכת
כי מעכשיו - כל מי שיתקשר למערכת דרך המספר 079512345 (לצור העניין) יעבור לשלוחה 4
ניתן לרכוש מספר נוסף דרך שירות הלקוחות

ניתוב למערכות אחרות של ימות המשיח לפי הגדרות זמנים (אין חיוב ביחידות)

type=routing_yemot
routing_time=yes
routing_time_1=077136000.09:00-17:00.mon-fri.1-20.jun
routing_time_2=0799234770.*.*.1-20.jun
routing_time_3=0773137770
להגדרות והסבר על הזמנים והגדרתם ניתן להסתכל ממודול routing_time, זה בדיוק אותו דבר
רק שיעבוד רק על מספרים נייחים של מערכות בימות המשיח בלבד, וללא חיוב ביחידות.

ברירת מחדל השיחות מוקלטות אם רוצים לבטל הקלטה יש להגדיר

routing_record=no
ההקלטות נשמרות בתוך התקיה הזמנית - בתוך הסל מחזור תחת תיקיית RoutingRecord וימחקו לאחר כ30 יום.

ניתוב שיחה לIP
type=routing_ip
routing_ip=11.22.33.44
routing_extension=0123456789
;routing_ip_dial_time=23
;routing_ip_cancel_star_next=yes
;routing_ip_end_goto=/8
;routing_ip_record=yes
;routing_ip_emeil_send=yes
;routing_ip_email_address=ss@hh.co.il
;routing_ip_email_name=אברהם
עידכון 1/6/2020: אין צורך באישור כתובות IP להפניית שיחות IP.

חובה להכניס מספר routing_extension כלומר מספר שלוחה לניתוב ביעד
לא ניתן לשלב תוים "@" "&" ":" בשלוחה.

ברירת מחדל החיוג מתבצע באורך 23 שניות
ניתן לשנות את אורך החיוג לפי הצורך בערך

routing_ip_dial_time=23
בחירת מחדל פורט הSIP היוצא הוא 5060.
ניתן להגדיר אחרת:

routing_ip_port=XX
ברירת מחדל שהשיחות לא מוקלטות
להקלטת השיחה יש להוסיף את הערך

routing_ip_record=yes
ברירת מחדל שגם אם השיחה מוקלטת אינה נשלחת למייל ואם רוצים שישלח יש להוסיף את הערך

routing_ip_emeil_send=yes
האימייל כברירת מחדל נשלח לכתובת המוגדרת בחייגן באם רוצים לשנות יש להכניס את האימייל לערך

routing_ip_email_address=ss@hh.co.il
באם רוצים להוסיף שם נוסף לאימייל בנוסף למוגדר בחייגן יש להוסיף לערך

routing_ip_email_name=אברהם
ברירת מחדל שבהקשת כוכבית השיחה מתנתקת וממשיכה במערכת הבסיס.
באם רוצים שלא יוכל לנתק במקש כוכבית (לדוגמא במערכת הבחירות במספרהמנותב היה צריך להקיש ת.ז. וכוכבית, בעוד שכוכבית מנתק השיחה) יש להכניס את הערך

routing_ip_cancel_star_next=yes
לאחר סיום ההשמעה לאיזה תיקיה לעבור - ברירת מחדל חוזר שלב אחורה
בדוגמא כאן הולך לשלוחה 8 שתחת שלוחה ראשית

routing_ip_end_goto=/8
זיהוי השיחה יוצאת
להלן האפשרויות:

routing_your_id=no01
routing_your_id=did
no01 - זיהוי רגיל של היוזם ללא תוספות
did - משדר מספר מערכת ולא מספר של הלקוח
במידה ומספר הלקוח מוגדר כחסום, המערכת תשדר את מספר המערכת של הלקוח במקום את המספר החסום שלו.

הגדרת זיהוי מיוחד
במודולים של
routing
routing_time
ניתן להגדיר מספרים נוספים לזיהוי
כל מספר שהוא מוגדר אצלכם במערכת כמספר משני - ניתן להגדיר גם אותו

במידת הצורך במידה ויש לכם מספר שהוא שייך לכם במערכת אחרת, ואתם רוצים לצאת עם הזיהוי שלו, גם ניתן
אבל חובה לעבור דרך שרות לקוחות שיאשרו למערכת שלכם להשתמש במספר
רק לאחר האישור ועדכון המספר הזיהוי שביקשתם יעבוד

בשביל להגדיר את מספר משני / מספר שאושר בשרות לקוחות יש להגדיר:

routing_your_id=special.0501234567

routing_email_send=cancel

routing_email_address=ddd@gmail.com

הגדרות מוזיקה בהמתנה
ברירת מחדל - המחייג שומע צליל חיוג
כאילו מוגדר כך:

music_on_hold=no
ניתן לשלוח לשירות הלקוחות מנגינה
שיגדירו אותה כמנגינה להמתנה
ואז מקבלים מהם שם המנגינה
וכותבים במקום no' את שם המוזיקה בהמתנה

מוסיקה בהמתנה למענה אנושי לבחירתכם
, הזכויות שמורות לחברת ימות המשיח.
אם יהיה ביקוש נוסיף שירים נוספים לספריית המוסיקה לבחירה.

מוסיקה בלבד.
music_on_hold=ztomao
מוסיקה עם פתיחת "שיחתכם חשובה לנו"
music_on_hold=ztomao+patiach
מעבר בסיום החיוג אם השיחה לא נענתה וכד'
אחרי כמה זמן יעבור לשלוחה אחרת - אם אין מענה

routing_end_time=20
בדוגמה מוגדר שאם המחייג לא קיבל מענה יעבור אחרי 20 שניות
