במודול זה ניתן ליצור קבצי tts ע"י הקלדה או הקלטה.
ערך חובה לשים בשלוחה

type=create_tts
מיקום שמירת הקובץ
ברירת מחדל שומר בתיקיה הנוכחית.

כדי שישמור בתיקיה אחרת (בדוגמה מופיע שלוחה 1/1), יש להגדיר

create_tts_in_folder=1/1
בשביל שתהיה אפשרות בחירת שלוחה:
M3969 "אנא הקישו את מספר השלוחה אליה תירצו להכניס את המלל, ובסיום הקישו סולמית"
יש להגדיר

create_tts_in_folder=tfr
קובץ ההקלטה יכנס לתיקית KeyToTtsRecord הנמצא בסל המחזור
כדי שיכנס לשלוחה אחרת (בדוגמה מופיע שלוחה 1/1), יש להגדיר

create_tts_record_save_in_folder=1/1
בחירת שם קובץ
ברירת מחדל, שומר בסדר עוקב את המספר הכי גבוה בשלוחה.
כדי שיתן אפשרות לבחירת שם הקובץ, יש להגדיר

create_tts_select_file_number=yes
ואז ישמיע תפריט M3970 "אנא הקישו את מספר ההודעה לשמירה ובסיום הקישו סולמית, לבחירת הודעת מערכת הקישו כוכבית ולאחר מכן את מספר ההודעה ובסיום הקישו סולמית".

(שימו לב, ניתן לבחור מספר הודעה רגילה כגון "245" וניתן לבחור הודעת מערכת ע"י הקשת כוכבית לפני הקשת מספר ההודעה.
לדוגמה *1000 והתוצאה תהיה M1000)

ברירת מחדל משמיע תפריט M3971 "לכתיבת ההודעה הקישו 1, להקלטת הודעה והמרתה לטקסט הקישו 2"

אפשרות כתיבה
כדי שהמאזין יעבור ישירות להקלדה ללא תפריט, יש להגדיר

create_tts_type=keyboard
לפרטים מלאים על אפשרויות המקלדת לחצו כאן

אפשרות הקלטה
כדי שהמאזין יעבור ישירות להקלטה ללא תפריט, יש להגדיר

create_tts_type=record
לפרטים מלאים על זיהוי דיבור והמרתו לטקסט לחצו כאן

ערכים ברירת מחדל:

ערך ראשון: שקט - 2 שניות
ערך שני: מינימום הקלטת הודעה - 2 שניות
ערך שלישי: מקסימום הקלטת הודעה - 20 שניות

כדי לשנות את הערכים, יש להגדיר create_tts_record_option ובין הערכים להכניס מקף אמצעי ?? - ??

לדוגמה:
כדי להגדיר 6 שניות שקט, 5 שניות מינימום הקלטת הודעה, ו 10 שניות מקסימום הקלטת הודעה
יש להגדיר

create_tts_record_option=6-5-10
ברירת מחדל משמיע M3973 "המלל שנקלט הוא.." ומשמיע את המלל שנקלט.
לאחר מכן משמיע תפריט:
M3974 "לשמיעה חוזרת הקישו 1, לשליחת ההודעה הקישו 2, להקלטה מחדש הקישו 3, לביטול ויציאה הקישו 4."
בשביל שידלג על התפריט ויעבור לאישור מידי, יש להגדיר

create_tts_record_ok=#
הוספת שם משתמש
בברירת מחדל לא מוסיף שם.
כדי שהמערכת תוסיף לפני ההודעה "הודעה מאת.. XX", יש להגדיר

create_tts_add_val_name=yes
המערכת לוקחת את שם המשתמש המוגדר בכניסה לפי ENTER ID (ברירת מחדל לפי טלפון)

מעבר לשלוחה אחרת
כאשר הקובץ נוצר בהצלחה
ברירת מחדל,
משמיע הודעה M3975 "ההודעה נקלטה בהצלחה" וחוזר שלב אחד אחורה.
כדי שילך לשלוחה אחרת (בדוגמה מופיע שלוחה 1/1 מהשלוחה הנוכחית), יש להגדיר

create_tts_end_goto=1/1
לניתוק, יש להגדיר

create_tts_end_goto=hangup
כאשר יש כישלון
ברירת מחדל,
משמיע הודעה M1080 "שגיאה" וחוזר שלב אחד אחורה.
כדי שילך לשלוחה אחרת (בדוגמה מופיע שלוחה 1/1 מהתפריט הראשי), יש להגדיר

create_tts_error_goto=/1/1
לניתוק, יש להגדיר

create_tts_error_goto=hangup
