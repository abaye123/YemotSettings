תוכלו להגדיר שכשלקוח נכנס לאחת השלוחות הוא יקבל אוטומטית הודעות טקסט אליו לפלאפון

send_sms=yes
sms_text=טקסט ההודעה בשורה אחת ללא אנטר
במידה ותרצו לשלוח הודעה למישהו ספציפי, למשל מנהל שרוצה לקבל סמס על כל אחד שנכנס לו לשלוחה

send_sms_to=0501234567
במידה שנרצה לשלוח את הסמס מזיהוי ספיישל/מספר משנה נגדיר:

send_sms_from=012345678
אם אנחנו רוצים שבתחילת הסמס יהיה כתוב בסוגריים מרובעות המספר של הלקוח שנכנס לשלוחה יש להגדיר:

sms_from_phone_in_text=yes
